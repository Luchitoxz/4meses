  <script>
    const phrases = [
      "Sos mi lugar en el mundo.",
      "Gracias por existir.",
      "Cada día te elijo.",
      "Tu sonrisa me cambia el día.",
      "No veo la hora de abrazarte.",
      "Estoy orgulloso de nosotros.",
      "Sos mi mejor casualidad.",
      "Amarte es lo más lindo que me pasó.",
      "Con vos aprendí a amar a la distancia.",
      "Sos más de lo que soñé.",
      "Nada me hace más feliz que vos.",
      "Nuestro amor crece cada día más.",
      "Todo es mejor desde que estás en mi vida."
    ];

    function showPhrase() {
      const random = Math.floor(Math.random() * phrases.length);
      document.getElementById('phrase').innerText = phrases[random];
    }

    function countDays() {
      const startDate = new Date("2025-02-05T00:00:00");
      const now = new Date();
      const diffTime = now - startDate;
      const diffDays = Math.floor(diffTime / (1000 * 60 * 60 * 24));
      document.getElementById('counter').innerText = `Desde el 5 de febrero de 2025, pasaron ${diffDays} días juntos 💖`;
    }

    function updateCountdown() {
      const now = new Date();
      const startDate = new Date("2025-02-05T00:00:00");
      const diff = now - startDate;

      const seconds = Math.floor(diff / 1000);
      const minutes = Math.floor(seconds / 60);
      const hours = Math.floor(minutes / 60);
      const days = Math.floor(hours / 24);

      const displayHours = hours % 24;
      const displayMinutes = minutes % 60;
      const displaySeconds = seconds % 60;

      document.getElementById("countdown").innerText = `Van ${days} días, ${displayHours} horas, ${displayMinutes} minutos y ${displaySeconds} segundos desde que empezó todo ✨`;
    }

    function createHearts() {
      const colors = ['#ff7b9c', '#ffb6c1', '#ffc0cb', '#ff4d6d'];
      for (let i = 0; i < 30; i++) {
        const heart = document.createElement('div');
        heart.classList.add('heart');
        heart.style.left = `${Math.random() * 100}vw`;
        heart.style.animationDuration = `${6 + Math.random() * 4}s`;
        heart.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
        document.body.appendChild(heart);
      }
    }

    countDays();
    updateCountdown();
    setInterval(updateCountdown, 1000);
    createHearts();
  </script>
